Archive backup:
run {
allocate channel 'dev_0' type 'sbt_tape' parms 'ENV=(OB2BARTYPE=Oracle8,OB2APPNAME=IDSPRD,OB2BARLIST=SEKIAZU0354_IDSPRD_ARCHIVE_BACKUP)';
crosscheck archivelog all;
backup
format 'SEKIAZU0354_IDSPRD_ARCHIVE_BACKUP<IDSPRD_%s:%t:%p>.dbf'
archivelog all FILESPERSET 10 DELETE INPUT;
}
