select dbms_lob.getlength(image) from HYBRIS.SYS_LOB0000014814C00017;


select object_name,object_type,owner from dba_objects where object_type='LOB' and owner not in ('SYS','XDB','SYSTEM','MDSYS');

select OWNER,TABLE_NAME from dba_lobs where owner not in ('SYS','XDB','SYSTEM','MDSYS');


COL owner	format a30
col table_name	format a30
col column_name	format a30
col segment_name format a40
col tablespace_name format a20
select 	l.owner owner, l.table_name table_name, l.column_name column_name, l.segment_name segment_name, l.tablespace_name tablespace_name, 
	l.cache cache, l.logging logging, l.in_row in_row, s.bytes/1024/1024/1024 GBytes
 from 	dba_lobs l, dba_segments s
where 	l.segment_name=s.segment_name
  and 	l.owner=s.owner
  and 	l.owner not in ('SYS','SYSTEM')
